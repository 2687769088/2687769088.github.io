## 题目：输入三个整数x,y,z，请把这三个数由小到大输出。
```java
def f(x,y,z):
    if x % 1 == 0 and y % 1 == 0 and z % 1 == 0:
       return True
a = input()
b = input()
c = input()
if f(a,b,c):
    if a > b:
        if a > c:
            if c > b:
                print(a>c>b)
            else:
                print(a>b>c)
        else:
            print(c>a>b)
    elif c > b and b > a:
        print(c>b>a)
    elif b > a and a > c:
        print(b>a>c)
    elif b > c and c > a:
        print(b>c>a)
```
